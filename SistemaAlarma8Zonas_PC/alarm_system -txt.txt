def get_base_path():
    """Obtiene la ruta base donde se almacenan los archivos de configuración"""
    if getattr(sys, 'frozen', False):
        # Si estamos ejecutando como .exe
        base_dir = os.path.dirname(sys.executable)
    else:
        # Si estamos ejecutando desde el código fuente
        base_dir = "."
    
    # Crear directorio de configuración si no existe
    config_dir = os.path.join(base_dir, "config")
    if not os.path.exists(config_dir):
        os.makedirs(config_dir, exist_ok=True)
    
    # Crear directorio de logs
    log_dir = os.path.join(base_dir, "logs")
    if not os.path.exists(log_dir):
        os.makedirs(log_dir, exist_ok=True)
    
    return base_dir, config_dir, log_dir

class AlarmState(Enum):
    """Estados posibles del sistema de alarma"""
    NORMAL = "SISTEMA NORMAL"
    ACTIVE = "¡ALARMA ACTIVADA!"
    PARTIAL = "ALARMA PARCIAL"
    DISABLED = "SISTEMA DESHABILITADO"

class AlarmEvent:
    """Evento de alarma con toda la información"""
    
    def __init__(self, event_type, zone=None, state=None, enabled=None, 
                 user=None, ip=None, zones_state=None, description=""):
        self.timestamp = datetime.now()
        self.event_type = event_type  # Tipo simple ahora
        self.zone = zone
        self.state = state
        self.enabled = enabled
        self.user = user
        self.ip = ip
        self.zones_state = zones_state or [1]*8
        self.description = description
        
    def to_dict(self):
        """Convierte el evento a diccionario para serialización"""
        return {
            'timestamp': self.timestamp.isoformat(),
            'event_type': self.event_type,
            'zone': self.zone,
            'state': self.state,
            'enabled': self.enabled,
            'user': self.user,
            'ip': self.ip,
            'zones_state': self.zones_state,
            'description': self.description
        }
    
    @classmethod
    def from_dict(cls, data):
        """Crea un evento desde diccionario"""
        event = cls(
            event_type=data['event_type'],
            zone=data['zone'],
            state=data['state'],
            enabled=data['enabled'],
            user=data['user'],
            ip=data['ip'],
            zones_state=data['zones_state'],
            description=data['description']
        )
        event.timestamp = datetime.fromisoformat(data['timestamp'])
        return event

class ZoneNamesManager:
    """Gestor de nombres personalizados para zonas"""
    
    def __init__(self):
        self.names_file = ZONE_NAMES_FILE
        self.zone_names = self.load_zone_names()
    
    def load_zone_names(self):
        """Carga los nombres de las zonas desde archivo"""
        default_names = {
            "0": "Zona 0",
            "1": "Zona 1", 
            "2": "Zona 2",
            "3": "Zona 3",
            "4": "Zona 4",
            "5": "Zona 5",
            "6": "Zona 6",
            "7": "Zona 7"
        }
        
        try:
            if os.path.exists(self.names_file):
                with open(self.names_file, 'r', encoding='utf-8') as f:
                    loaded_names = json.load(f)
                    # Asegurarse de que tenemos nombres para todas las zonas
                    for i in range(8):
                        if str(i) not in loaded_names:
                            loaded_names[str(i)] = default_names[str(i)]
                    return loaded_names
            else:
                # Guardar nombres por defecto
                self.save_zone_names(default_names)
                return default_names
        except Exception as e:
            print(f"Error cargando nombres de zonas: {e}")
            return default_names
    
    def save_zone_names(self, names=None):
        """Guarda los nombres de las zonas en archivo"""
        try:
            if names is None:
                names = self.zone_names
            
            with open(self.names_file, 'w', encoding='utf-8') as f:
                json.dump(names, f, indent=2, ensure_ascii=False)
        except Exception as e:
            print(f"Error guardando nombres de zonas: {e}")
    
    def get_zone_name(self, zone_number):
        """Obtiene el nombre de una zona específica"""
        return self.zone_names.get(str(zone_number), f"Zona {zone_number}")
    
    def set_zone_name(self, zone_number, name):
        """Establece un nuevo nombre para una zona"""
        self.zone_names[str(zone_number)] = name
        self.save_zone_names()
    
    def reset_zone_name(self, zone_number):
        """Restablece el nombre por defecto de una zona"""
        default_name = f"Zona {zone_number}"
        self.zone_names[str(zone_number)] = default_name
        self.save_zone_names()
        return default_name

class AlarmStateManager:
    """Gestor del estado y historial de alarmas"""
    
    def __init__(self, security_mgr=None):
        self.security_mgr = security_mgr
        self.history_file = ALARM_HISTORY_FILE
        self.state_file = ALARM_STATE_FILE
        self.alarm_history = []
        self.current_alarm_state = AlarmState.NORMAL
        
        # Cargar historial y estado previo
        self.load_alarm_state()
        self.load_alarm_history()
        
        # Inicializar con todas las alarmas habilitadas y en estado normal
        self.zones_state = [1] * 8  # 1 = Normal, 0 = Abierta
        self.zones_enabled = [1] * 8  # 1 = Habilitada, 0 = Deshabilitada
        
    def save_alarm_state(self):
        """Guarda el estado actual de las alarmas"""
        try:
            state_data = {
                'current_alarm_state': self.current_alarm_state.value,
                'zones_state': self.zones_state,
                'zones_enabled': self.zones_enabled,
                'last_update': datetime.now().isoformat()
            }
            
            with open(self.state_file, 'w', encoding='utf-8') as f:
                json.dump(state_data, f, indent=2, ensure_ascii=False)
                
        except Exception as e:
            print(f"Error guardando estado de alarmas: {e}")
    
    def load_alarm_state(self):
        """Carga el estado anterior de las alarmas"""
        try:
            if os.path.exists(self.state_file):
                with open(self.state_file, 'r', encoding='utf-8') as f:
                    state_data = json.load(f)
                    
                self.current_alarm_state = AlarmState(state_data['current_alarm_state'])
                self.zones_state = state_data.get('zones_state', [1]*8)
                self.zones_enabled = state_data.get('zones_enabled', [1]*8)
                
                # Asegurar que tenemos 8 zonas
                if len(self.zones_state) < 8:
                    self.zones_state = (self.zones_state + [1]*(8 - len(self.zones_state)))[:8]
                if len(self.zones_enabled) < 8:
                    self.zones_enabled = (self.zones_enabled + [1]*(8 - len(self.zones_enabled)))[:8]
                    
        except Exception as e:
            print(f"Error cargando estado de alarmas: {e}")
            # Inicializar por defecto
            self.zones_state = [1] * 8
            self.zones_enabled = [1] * 8
            self.current_alarm_state = AlarmState.NORMAL
    
    def save_alarm_history(self):
        """Guarda el historial de alarmas en CSV"""
        try:
            # Asegurarse de que tenemos datos para guardar
            if not self.alarm_history:
                print("No hay historial para guardar")
                return
                
            with open(self.history_file, 'w', newline='', encoding='utf-8') as csvfile:
                fieldnames = ['timestamp', 'event_type', 'zone', 'state', 'enabled', 
                            'user', 'ip', 'zones_state', 'description']
                writer = csv.DictWriter(csvfile, fieldnames=fieldnames)
                
                writer.writeheader()
                for event in self.alarm_history[-1000:]:  # Limitar a últimos 1000 eventos
                    try:
                        writer.writerow(event.to_dict())
                    except Exception as e:
                        print(f"Error escribiendo evento: {e}")
                        continue
                        
        except Exception as e:
            print(f"Error guardando historial de alarmas: {e}")
    
    def load_alarm_history(self):
        """Carga el historial de alarmas desde CSV"""
        try:
            if os.path.exists(self.history_file):
                self.alarm_history = []
                with open(self.history_file, 'r', encoding='utf-8') as csvfile:
                    reader = csv.DictReader(csvfile)
                    if reader.fieldnames is None:
                        print("Archivo de historial vacío o corrupto")
                        return
                        
                    for row in reader:
                        try:
                            # Convertir zones_state de string a lista
                            zones_state = [1]*8
                            if 'zones_state' in row and row['zones_state']:
                                try:
                                    zones_state = json.loads(row['zones_state'])
                                    if isinstance(zones_state, str):
                                        zones_state = [int(c) for c in zones_state if c.isdigit()]
                                    elif isinstance(zones_state, list):
                                        zones_state = [int(s) for s in zones_state[:8]]
                                except:
                                    zones_state = [1]*8
                            
                            event = AlarmEvent(
                                event_type=row.get('event_type', ''),
                                zone=int(row['zone']) if row.get('zone') and row['zone'].strip() else None,
                                state=int(row['state']) if row.get('state') and row['state'].strip() else None,
                                enabled=int(row['enabled']) if row.get('enabled') and row['enabled'].strip() else None,
                                user=row.get('user', ''),
                                ip=row.get('ip', ''),
                                zones_state=zones_state,
                                description=row.get('description', '')
                            )
                            event.timestamp = datetime.fromisoformat(row['timestamp'])
                            self.alarm_history.append(event)
                        except Exception as e:
                            print(f"Error procesando fila del historial: {e}")
                            continue
                            
                print(f"Cargados {len(self.alarm_history)} eventos del historial")
            else:
                print("Archivo de historial no existe, se creará uno nuevo")
                self.alarm_history = []
                
        except Exception as e:
            print(f"Error cargando historial de alarmas: {e}")
            self.alarm_history = []
    
    def add_alarm_event(self, event_type, zone=None, state=None, enabled=None, 
                       description="", ip=None, zones_state=None):
        """Agrega un evento al historial"""
        user = self.security_mgr.current_user if self.security_mgr else None
        zones_state = zones_state or self.zones_state.copy()
        
        # Usar tipos de evento simples
        simple_event_types = {
            "ALARM_TRIGGERED": "Alarma",
            "ZONE_STATE_CHANGE": "Estado",
            "ZONE_ENABLE_CHANGE": "Habilitar",
            "USER_LOGIN": "Login",
            "ALARM_STATE_CHANGE": "Sistema",
            "USER_LOGOUT": "Logout",
            "SYSTEM_SHUTDOWN": "Apagado",
            "CONNECTION_LOST": "Conexión",
            "AUTO_RECONNECT": "Reconexión",
            "ZONE_NAME_CHANGE": "Nombre"
        }
        
        simple_type = simple_event_types.get(event_type, event_type)
        
        event = AlarmEvent(
            event_type=simple_type,
            zone=zone,
            state=state,
            enabled=enabled,
            user=user,
            ip=ip,
            zones_state=zones_state,
            description=description
        )
        
        self.alarm_history.append(event)
        self.save_alarm_history()
        
        # Actualizar estado de zona si es relevante
        if zone is not None and 0 <= zone <= 7:
            if state is not None:
                self.zones_state[zone] = state
            if enabled is not None:
                self.zones_enabled[zone] = enabled
        
        # Actualizar estado global de alarma
        self.update_alarm_state()
        
        # Guardar estado actual
        self.save_alarm_state()
        
        print(f"Evento agregado al historial: {simple_type} - {description}")
        
        return event
    
    def update_alarm_state(self):
        """Actualiza el estado global de alarma basado en las zonas"""
        # Verificar si hay al menos una zona en alarma (estado 0) y habilitada
        any_alarm_active = False
        
        for i in range(8):
            if self.zones_enabled[i] == 1 and self.zones_state[i] == 0:
                any_alarm_active = True
                break
        
        # Actualizar estado global
        if any_alarm_active:
            self.current_alarm_state = AlarmState.ACTIVE
        else:
            self.current_alarm_state = AlarmState.NORMAL
        
        # Registrar cambio de estado
        if self.security_mgr:
            self.security_mgr.log_event(
                "ALARM_STATE_CHANGE",
                f"Estado cambiado a: {self.current_alarm_state.value}"
            )
    
    def get_alarm_state_text(self):
        """Obtiene el texto del estado actual de alarma"""
        return self.current_alarm_state.value
    
    def get_recent_events(self, limit=100):
        """Obtiene los eventos más recientes"""
        return self.alarm_history[-limit:] if self.alarm_history else []
    
    def clear_history(self):
        """Limpia el historial de alarmas"""
        self.alarm_history = []
        self.save_alarm_history()

class SecurityManager:
    """Gestor de seguridad, usuarios y logging"""
    
    def __init__(self):
        self.users_file = USERS_FILE
        self.crypto_enabled = CRYPTO_AVAILABLE
        if self.crypto_enabled:
            self.key = self._load_or_create_key()
            self.cipher = Fernet(self.key)
        self.logger = self._setup_logger()
        self.current_user = None
        
    def _load_or_create_key(self):
        """Carga o crea clave de cifrado"""
        if not self.crypto_enabled:
            return None
            
        key_file = os.path.join(CONFIG_DIR, "secret.key")
        if os.path.exists(key_file):
            with open(key_file, 'rb') as f:
                return f.read()
        else:
            key = Fernet.generate_key()
            with open(key_file, 'wb') as f:
                f.write(key)
            return key
    
    def _setup_logger(self):
        """Configura sistema de logging"""
        logger = logging.getLogger('AlarmSystem')
        logger.setLevel(logging.INFO)
        
        # Formato del log
        formatter = logging.Formatter(
            '%(asctime)s - %(levelname)s - %(message)s',
            datefmt='%Y-%m-%d %H:%M:%S'
        )
        
        # Handler para archivo
        file_handler = logging.FileHandler(LOG_FILE)
        file_handler.setFormatter(formatter)
        logger.addHandler(file_handler)
        
        # Handler para consola
        console_handler = logging.StreamHandler()
        console_handler.setFormatter(formatter)
        logger.addHandler(console_handler)
        
        return logger
    
    def register_user(self, username, password, role="operator"):
        """Registra un nuevo usuario"""
        users = self._load_users()
        
        if username in users:
            return False, "Usuario ya existe"
        
        # Hash de la contraseña con salt
        salt = os.urandom(32)
        key = hashlib.pbkdf2_hmac(
            'sha256',
            password.encode('utf-8'),
            salt,
            100000
        )
        
        users[username] = {
            'salt': salt.hex(),
            'key': key.hex(),
            'role': role,
            'created': datetime.now().isoformat(),
            'last_login': None
        }
        
        self._save_users(users)
        self.logger.info(f"Usuario registrado: {username} - Rol: {role}")
        return True, "Usuario registrado exitosamente"
    
    def authenticate(self, username, password):
        """Autentica un usuario"""
        users = self._load_users()
        
        if username not in users:
            return False, "Usuario no encontrado"
        
        user_data = users[username]
        salt = bytes.fromhex(user_data['salt'])
        stored_key = bytes.fromhex(user_data['key'])
        
        # Verificar contraseña
        key = hashlib.pbkdf2_hmac(
            'sha256',
            password.encode('utf-8'),
            salt,
            100000
        )
        
        if key == stored_key:
            # Actualizar último login
            user_data['last_login'] = datetime.now().isoformat()
            users[username] = user_data
            self._save_users(users)
            
            self.current_user = username
            self.logger.info(f"Login exitoso: {username}")
            return True, "Autenticación exitosa"
        
        self.logger.warning(f"Intento fallido de login: {username}")
        return False, "Contraseña incorrecta"
    
    def is_admin(self):
        """Verifica si el usuario actual es administrador"""
        if not self.current_user:
            return False
            
        users = self._load_users()
        user_data = users.get(self.current_user)
        
        if user_data and user_data.get('role') == 'admin':
            return True
        return False
    
    def log_event(self, event_type, message, device_ip=None):
        """Registra evento en el log"""
        user_info = f" - Usuario: {self.current_user}" if self.current_user else ""
        device_info = f" - Dispositivo: {device_ip}" if device_ip else ""
        full_message = f"{event_type}: {message}{user_info}{device_info}"
        self.logger.info(full_message)
    
    def _load_users(self):
        """Carga usuarios desde archivo"""
        if not os.path.exists(self.users_file):
            return {}
        
        try:
            with open(self.users_file, 'r', encoding='utf-8') as f:
                if self.crypto_enabled:
                    encrypted_data = f.read()
                    decrypted_data = self.cipher.decrypt(encrypted_data.encode())
                    return json.loads(decrypted_data)
                else:
                    return json.load(f)
        except Exception as e:
            print(f"Error cargando usuarios: {e}")
            return {}
    
    def _save_users(self, users):
        """Guarda usuarios en archivo"""
        try:
            data = json.dumps(users, indent=2, ensure_ascii=False)
            
            if self.crypto_enabled:
                encrypted_data = self.cipher.encrypt(data.encode())
                with open(self.users_file, 'wb') as f:
                    f.write(encrypted_data)
            else:
                with open(self.users_file, 'w', encoding='utf-8') as f:
                    f.write(data)
        except Exception as e:
            print(f"Error guardando usuarios: {e}")

class AlarmDevice:
    """Representa un dispositivo de alarma remoto con reconexión automática"""
    
    def __init__(self, ip, port=DEFAULT_PORT, security_mgr=None, alarm_state_mgr=None):
        self.ip = ip
        self.port = port
        self.socket = None
        self.connected = False
        self.receive_thread = None
        self.receive_queue = queue.Queue()
        self.command_queue = queue.Queue()
        self.security_mgr = security_mgr
        self.alarm_state_mgr = alarm_state_mgr
        self.status = {
            'zones': [1]*8,  # 1=normal, 0=alarma
            'enabled': [1]*8,  # 1=habilitada, 0=deshabilitada
            'alarm_active': 0,
            'raw_state': 0xFF,
            'enabled_mask': 0xFF
        }
        self.last_update = None
        self.reconnect_thread = None
        self.auto_reconnect = True
        self.stop_reconnect = threading.Event()
        
        # Cargar estado guardado para este dispositivo
        if alarm_state_mgr:
            self.status['zones'] = alarm_state_mgr.zones_state.copy()
            self.status['enabled'] = alarm_state_mgr.zones_enabled.copy()
            self.status['alarm_active'] = 1 if alarm_state_mgr.current_alarm_state == AlarmState.ACTIVE else 0
        
    def connect(self):
        """Establece conexión con el dispositivo"""
        try:
            self.socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            self.socket.settimeout(5)
            self.socket.connect((self.ip, self.port))
            
            # Configurar timeout para recepción
            self.socket.settimeout(0.1)
            
            self.connected = True
            self.stop_reconnect.clear()
            
            # Iniciar hilo de recepción
            self.receive_thread = threading.Thread(target=self._receive_loop, daemon=True)
            self.receive_thread.start()
            
            # Iniciar hilo de procesamiento de comandos
            command_thread = threading.Thread(target=self._command_loop, daemon=True)
            command_thread.start()
            
            # Iniciar hilo de verificación de conexión
            self.reconnect_thread = threading.Thread(target=self._connection_check_loop, daemon=True)
            self.reconnect_thread.start()
            
            if self.security_mgr:
                self.security_mgr.log_event("CONEXION", f"Conectado a {self.ip}:{self.port}", self.ip)
            
            # Enviar estado inicial de zonas al dispositivo
            self._sync_zones_to_device()
            
            return True, "Conexión exitosa"
            
        except socket.timeout:
            return False, "Timeout de conexión"
        except ConnectionRefusedError:
            return False, "Conexión rechazada"
        except Exception as e:
            return False, f"Error de conexión: {str(e)}"
    
    def _connection_check_loop(self):
        """Verifica periódicamente la conexión y reconecta si es necesario"""
        while not self.stop_reconnect.is_set() and self.auto_reconnect:
            time.sleep(2)  # Verificar cada 2 segundos
            
            if not self.connected:
                try:
                    print(f"Intentando reconexión a {self.ip}:{self.port}...")
                    
                    # Intentar reconectar
                    self.socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
                    self.socket.settimeout(5)
                    self.socket.connect((self.ip, self.port))
                    self.socket.settimeout(0.1)
                    
                    self.connected = True
                    
                    # Reiniciar hilos
                    self.receive_thread = threading.Thread(target=self._receive_loop, daemon=True)
                    self.receive_thread.start()
                    
                    command_thread = threading.Thread(target=self._command_loop, daemon=True)
                    command_thread.start()
                    
                    if self.alarm_state_mgr:
                        self.alarm_state_mgr.add_alarm_event(
                            event_type="AUTO_RECONNECT",
                            description=f"Reconexión automática exitosa a {self.ip}",
                            ip=self.ip
                        )
                    
                    print(f"Reconexión exitosa a {self.ip}:{self.port}")
                    
                except Exception as e:
                    print(f"Error en reconexión: {e}")
                    if self.socket:
                        self.socket.close()
                        self.socket = None
                    self.connected = False
    
    def _sync_zones_to_device(self):
        """Sincroniza el estado de zonas con el dispositivo"""
        if not self.connected:
            return
        
        # Habilitar/deshabilitar zonas según el estado guardado
        for i in range(8):
            if self.status['enabled'][i] == 1:
                self.command_queue.put(f"E {i}")
            else:
                self.command_queue.put(f"D {i}")
    
    def disconnect(self):
        """Cierra la conexión"""
        self.connected = False
        self.auto_reconnect = False
        self.stop_reconnect.set()
        
        if self.socket:
            self.socket.close()
            self.socket = None
        
        if self.security_mgr:
            self.security_mgr.log_event("DESCONEXION", f"Desconectado de {self.ip}", self.ip)
    
    def send_command(self, command):
        """Envía un comando al dispositivo"""
        if not self.connected:
            # Intentar reconectar automáticamente
            if self.auto_reconnect:
                return False, "Intentando reconectar..."
            return False, "No conectado"
        
        try:
            full_command = command + "\n"
            self.socket.send(full_command.encode('utf-8'))
            
            if self.security_mgr:
                self.security_mgr.log_event("COMANDO", f"Enviado: {command.strip()}", self.ip)
            
            return True, "Comando enviado"
        except Exception as e:
            self.connected = False
            return False, f"Error al enviar: {str(e)}"
    
    def _receive_loop(self):
        """Hilo para recibir datos del dispositivo"""
        buffer = ""
        
        while self.connected and not self.stop_reconnect.is_set():
            try:
                data = self.socket.recv(1024)
                if data:
                    buffer += data.decode('utf-8', errors='ignore')
                    
                    # Procesar líneas completas
                    while '\n' in buffer:
                        line, buffer = buffer.split('\n', 1)
                        line = line.strip()
                        if line:
                            self._process_received_line(line)
                            
            except socket.timeout:
                continue
            except Exception as e:
                print(f"Error en recepción: {e}")
                self.connected = False
                
                # Registrar pérdida de conexión
                if self.alarm_state_mgr:
                    self.alarm_state_mgr.add_alarm_event(
                        event_type="CONNECTION_LOST",
                        description=f"Conexión perdida con {self.ip}",
                        ip=self.ip
                    )
                break
    
    def _process_received_line(self, line):
        """Procesa una línea recibida del dispositivo"""
        # Registrar evento
        if self.security_mgr:
            self.security_mgr.log_event("RECIBIDO", line, self.ip)
        
        # Procesar diferentes tipos de mensajes
        if line.startswith("ALARM 8Z"):
            print(f"Dispositivo: {line}")
            
        elif line.startswith("READY"):
            print("Dispositivo listo")
            
        elif line.startswith("S "):
            self._parse_status(line)
            
        elif line.startswith("Z"):
            self._parse_zone_event(line)
            
        elif line.startswith("!"):
            self._parse_alarm_event(line)
            
        elif line.startswith("R OK"):
            self.status['alarm_active'] = 0
            if self.alarm_state_mgr:
                self.alarm_state_mgr.update_alarm_state()
            
        elif line.startswith("E Z") or line.startswith("D Z"):
            self._parse_enable_disable(line)
        
        # Poner en cola para la interfaz
        self.receive_queue.put(line)
        self.last_update = datetime.now()
    
    def _parse_status(self, line):
        """Parsea mensaje de estado"""
        # Formato: S 0xFF E:0xFF A:0
        try:
            parts = line.split()
            if len(parts) >= 4:
                raw_state = int(parts[1], 16)
                enabled_mask = int(parts[2].split(':')[1], 16)
                alarm_active = int(parts[3].split(':')[1])
                
                self.status['raw_state'] = raw_state
                self.status['enabled_mask'] = enabled_mask
                self.status['alarm_active'] = alarm_active
                
                # Actualizar estado de zonas individuales
                for i in range(8):
                    zone_state = (raw_state >> i) & 1
                    zone_enabled = (enabled_mask >> i) & 1
                    self.status['zones'][i] = zone_state
                    self.status['enabled'][i] = zone_enabled
                
                # Actualizar el estado de alarma
                if self.alarm_state_mgr:
                    # Sincronizar estado con el manager
                    self.alarm_state_mgr.zones_state = self.status['zones'].copy()
                    self.alarm_state_mgr.zones_enabled = self.status['enabled'].copy()
                    self.alarm_state_mgr.update_alarm_state()
                    
        except Exception as e:
            print(f"Error parseando estado: {e}")
    
    def _parse_zone_event(self, line):
        """Parsea evento de zona (ej: Z3:0)"""
        try:
            zone = int(line[1])
            state = int(line[3])
            
            if 0 <= zone <= 7:
                old_state = self.status['zones'][zone]
                self.status['zones'][zone] = state
                
                # Registrar evento de cambio de zona
                if self.alarm_state_mgr:
                    event_type = "ZONE_STATE_CHANGE"
                    description = f"Zona {zone} cambió de {old_state} a {state}"
                    self.alarm_state_mgr.add_alarm_event(
                        event_type=event_type,
                        zone=zone,
                        state=state,
                        enabled=self.status['enabled'][zone],
                        description=description,
                        ip=self.ip
                    )
                
        except:
            pass
    
    def _parse_alarm_event(self, line):
        """Parsea evento de alarma (ej: !Z3)"""
        try:
            zone = int(line[2])
            if 0 <= zone <= 7:
                self.status['zones'][zone] = 0
                self.status['alarm_active'] = 1
                
                # Registrar evento de alarma
                if self.alarm_state_mgr:
                    event_type = "ALARM_TRIGGERED"
                    description = f"Alarma activada en zona {zone}"
                    self.alarm_state_mgr.add_alarm_event(
                        event_type=event_type,
                        zone=zone,
                        state=0,
                        enabled=self.status['enabled'][zone],
                        description=description,
                        ip=self.ip
                    )
                
        except:
            pass
    
    def _parse_enable_disable(self, line):
        """Parsea habilitación/deshabilitación de zona"""
        try:
            zone = int(line[3])
            enabled = 1 if line.startswith("E") else 0
            
            if 0 <= zone <= 7:
                old_enabled = self.status['enabled'][zone]
                self.status['enabled'][zone] = enabled
                
                # Actualizar máscara
                mask = 0
                for i in range(8):
                    if self.status['enabled'][i]:
                        mask |= (1 << i)
                self.status['enabled_mask'] = mask
                
                # Registrar evento de habilitación/deshabilitación
                if self.alarm_state_mgr:
                    event_type = "ZONE_ENABLE_CHANGE"
                    description = f"Zona {zone} {'habilitada' if enabled else 'deshabilitada'}"
                    self.alarm_state_mgr.add_alarm_event(
                        event_type=event_type,
                        zone=zone,
                        state=self.status['zones'][zone],
                        enabled=enabled,
                        description=description,
                        ip=self.ip
                    )
                
        except:
            pass
    
    def _command_loop(self):
        """Procesa comandos en cola"""
        while self.connected and not self.stop_reconnect.is_set():
            try:
                command = self.command_queue.get(timeout=0.5)
                success, message = self.send_command(command)
                if not success:
                    self.receive_queue.put(f"ERROR: {message}")
            except queue.Empty:
                continue
    
    def get_zone_status(self, zone):
        """Obtiene estado de una zona específica"""
        if 0 <= zone <= 7:
            return {
                'state': self.status['zones'][zone],
                'enabled': self.status['enabled'][zone],
                'in_alarm': self.status['zones'][zone] == 0 and self.status['enabled'][zone] == 1
            }
        return None

def center_window(window, width=None, height=None):
    """Centra una ventana en la pantalla"""
    window.update_idletasks()
    
    if width is None:
        width = window.winfo_width()
    if height is None:
        height = window.winfo_height()
    
    x = (window.winfo_screenwidth() // 2) - (width // 2)
    y = (window.winfo_screenheight() // 2) - (height // 2)
    
    window.geometry(f'{width}x{height}+{x}+{y}')

            bg=COLORS['normal'],

    def load_history(self, filtered=False):
        """Carga el historial en el Treeview"""
        # Limpiar treeview
        for item in self.tree.get_children():
            self.tree.delete(item)
        
        # Obtener eventos
        if filtered and hasattr(self, 'filtered_events'):
            events = self.filtered_events
        else:
            events = self.alarm_state_mgr.get_recent_events(limit=100)
        
        # Insertar eventos
        for event in events:
            # Formatear valores para display
            zone_str = str(event.zone) if event.zone is not None else ""
            
            # Estado simplificado
            if event.state == 0:
                state_str = "Abierta"
            elif event.state == 1:
                state_str = "Normal"
            else:
                state_str = ""
            
            enabled_str = "Sí" if event.enabled == 1 else "No" if event.enabled == 0 else ""
            
            values = (
                event.timestamp.strftime("%Y-%m-%d %H:%M:%S"),
                event.event_type,  # Ya es simple
                zone_str,
                state_str,
                enabled_str,
                event.user or "",
                event.ip or "",
                event.description
            )
            
            self.tree.insert('', 'end', values=values)
        
        # Actualizar etiqueta de información
        count = len(events)
        self.info_label.config(text=f"Mostrando {count} eventos")
    
    def apply_filter(self):
        """Aplica los filtros seleccionados"""
        event_type = self.event_type_var.get()
        zone = self.zone_var.get()
        
        # Obtener todos los eventos
        all_events = self.alarm_state_mgr.get_recent_events(limit=1000)
        filtered_events = []
        
        for event in all_events:
            # Filtrar por tipo de evento
            if event_type != "Todos" and event.event_type != event_type:
                continue
            
            # Filtrar por zona
            if zone != "Todas":
                if event.zone is None:
                    continue
                if str(event.zone) != zone:
                    continue
            
            filtered_events.append(event)
        
        # Guardar eventos filtrados y cargarlos
        self.filtered_events = filtered_events
        self.load_history(filtered=True)
    
    def clear_filter(self):
        """Limpia los filtros"""
        self.event_type_var.set("Todos")
        self.zone_var.set("Todas")
        self.load_history()
    
    def export_to_csv(self):
        """Exporta el historial a CSV"""
        try:
            filename = f"alarm_history_export_{datetime.now().strftime('%Y%m%d_%H%M%S')}.csv"
            
            events = self.alarm_state_mgr.get_recent_events(limit=1000)
            
            with open(filename, 'w', newline='', encoding='utf-8') as csvfile:
                fieldnames = ['timestamp', 'event_type', 'zone', 'state', 
                            'enabled', 'user', 'ip', 'description']
                writer = csv.DictWriter(csvfile, fieldnames=fieldnames)
                
                writer.writeheader()
                for event in events:
                    writer.writerow({
                        'timestamp': event.timestamp.strftime("%Y-%m-%d %H:%M:%S"),
                        'event_type': event.event_type,
                        'zone': event.zone if event.zone is not None else "",
                        'state': event.state if event.state is not None else "",
                        'enabled': event.enabled if event.enabled is not None else "",
                        'user': event.user or "",
                        'ip': event.ip or "",
                        'description': event.description
                    })
            
            messagebox.showinfo("Éxito", f"Historial exportado a {filename}")
        except Exception as e:
            messagebox.showerror("Error", f"No se pudo exportar: {str(e)}")
    
    def clear_history(self):
        """Limpia el historial de alarmas - solo para admin"""
        if not self.security_mgr.is_admin():
            messagebox.showerror("Acceso Denegado", "Solo los administradores pueden limpiar el historial.")
            return
            
        if messagebox.askyesno("Confirmar", "¿Está seguro de que desea borrar todo el historial?"):
            self.alarm_state_mgr.clear_history()
            self.load_history()
            messagebox.showinfo("Éxito", "Historial limpiado correctamente")

    def toggle_connection(self):
        """Conecta o desconecta del dispositivo"""
        if self.current_device and self.current_device.connected:
            # Desconectar
            self.current_device.disconnect()
            self.current_device = None
            self.update_connection_status(False)
        else:
            # Conectar
            ip = self.ip_entry.get().strip()
            port = self.port_entry.get().strip()
            
            if not ip:
                messagebox.showerror("Error", "Debe especificar una dirección IP")
                return
            
            try:
                port = int(port)
            except ValueError:
                messagebox.showerror("Error", "Puerto inválido")
                return
            
            # Crear dispositivo
            device = AlarmDevice(ip, port, self.security_mgr, self.alarm_state_mgr)
            success, message = device.connect()
            
            if success:
                self.current_device = device
                self.devices[ip] = device
                self.update_connection_status(True)
                self.send_command("S")  # Solicitar estado inicial
            else:
                messagebox.showerror("Error de conexión", message)
    
    def update_connection_status(self, connected):
        """Actualiza la UI según estado de conexión"""
        if connected:
            self.connect_btn.config(
                text="DESCONECTAR",
                bg=COLORS['warning']
            )
            self.connection_status.config(
                text="Conectado",
                fg=COLORS['normal']
            )
        else:
            self.connect_btn.config(

                bg=COLORS['normal']
            )
            self.connection_status.config(
                text="Desconectado",
                fg=COLORS['disabled']

    def send_command(self, command):
        """Envía un comando al dispositivo actual"""
        if not self.current_device or not self.current_device.connected:
            messagebox.showwarning("Advertencia", "No hay dispositivo conectado")
            return
        
        self.current_device.command_queue.put(command)
    
    def enable_zone(self, zone):
        """Habilita una zona específica"""
        self.send_command(f"E {zone}")
        
        # Registrar evento localmente
        self.alarm_state_mgr.add_alarm_event(
            event_type="ZONE_ENABLE_CHANGE",
            zone=zone,
            enabled=1,
            description=f"Zona {zone} habilitada manualmente"
        )
    
    def disable_zone(self, zone):
        """Deshabilita una zona específica"""
        self.send_command(f"D {zone}")
        
        # Registrar evento localmente
        self.alarm_state_mgr.add_alarm_event(
            event_type="ZONE_ENABLE_CHANGE",
            zone=zone,
            enabled=0,
            description=f"Zona {zone} deshabilitada manualmente"
        )
    
    def update_ui_from_device(self):
        """Actualiza la UI con el estado del dispositivo"""
        if not self.current_device or not self.current_device.connected:
            return
        
        # Procesar mensajes recibidos
        while not self.current_device.receive_queue.empty():
            try:
                self.current_device.receive_queue.get_nowait()
            except queue.Empty:
                break
        
        # Actualizar estado de zonas
        for i in range(8):
            status = self.current_device.get_zone_status(i)
            if status:
                widget = self.zone_widgets[i]
                
                # Actualizar estado
                if status['in_alarm']:
                    widget['state_label'].config(
                        text="ABIERTA",
                        fg=COLORS['alarm']
                    )
                    widget['frame'].config(bg='#FFCDD2')  # Rojo claro
                else:
                    state_text = "NORMAL" if status['state'] == 1 else "ABIERTA"
                    color = COLORS['normal'] if status['state'] == 1 else COLORS['warning']
                    widget['state_label'].config(
                        text=state_text,
                        fg=color
                    )
                    widget['frame'].config(bg='white')
                
                # Actualizar habilitación
                if status['enabled']:
                    widget['enabled_label'].config(
                        text="HABILITADA",
                        fg=COLORS['enabled']
                    )
                else:
                    widget['enabled_label'].config(
                        text="DESHABILITADA",
                        fg=COLORS['disabled']
                    )
    
    def start_update_loop(self):
        """Inicia el loop de actualización de UI"""
        self.update_ui_from_device()
        self.root.after(100, self.start_update_loop)
    
    def logout(self):
        """Cierra sesión y vuelve al login"""
        if self.current_device and self.current_device.connected:
            self.current_device.disconnect()
        
        # Registrar evento de logout
        self.alarm_state_mgr.add_alarm_event(
            event_type="USER_LOGOUT",
            description=f"Usuario {self.security_mgr.current_user} cerró sesión"
        )
        
        self.security_mgr.current_user = None
        self.root.destroy()
        
        # Reiniciar aplicación
        security_mgr = SecurityManager()
        alarm_state_mgr = AlarmStateManager(security_mgr)
        login = LoginWindow(security_mgr, alarm_state_mgr)
        login.run()
    
    def on_closing(self):
        """Maneja el cierre de la ventana"""
        if messagebox.askokcancel("Salir", "¿Está seguro de que desea salir?"):
            if self.current_device and self.current_device.connected:
                self.current_device.disconnect()

            self.alarm_state_mgr.add_alarm_event(

                description="Aplicación cerrada"
            )
            
            self.security_mgr.log_event("SISTEMA", "Aplicación cerrada")
            self.root.destroy()
            sys.exit(0)
